Set horizontal and vertical coordinate labels

xlabel('Frequency','FontSize',16);
%Set the abscissa as 'frequency', font size: 16
ylabel('$|sinc(f*T_{coh})|$','Interpreter','latex','FontSize',18);
%Set the ordinate to the specified formula, the syntax interpreter is latex, font size: 18
xlim([-200 200]),ylim([0 1]);
%Set the vertical coordinate range to 0~1, and the horizontal coordinate range to -200~200;

Set horizontal and vertical coordinate scales

set(gca,'FontSize',14,'FontName','Times New Roman');
%Set horizontal and vertical coordinates font, size

Set the legend label

legend({'label 1','label 2','label 3'});
% set the label
legend('boxoff');
%Close the label box

Scale line and outline control

grid on;
%Open the tick marks in the drawing area
grid minor;
%Open the fine tick marks in the drawing area

box off;
%Close the drawing outline
axis off;
%Close the horizontal and vertical scale marks

Image annotation and annotation hiding

plot(x,'-o','MarkerIndices',[520 770],'MarkerEdgeColor','r','LineWidth',1.5);
% Draw the curve of x, and mark the 520th and 770th points with red circles

set(gca,'xtick',[],'xticklabel',[]); % hide the label of the x-axis scale
set(gca,'ytick',[],'yticklabel',[]); % hide the label of the y-axis scale
axis off; % turn off the axis and background

Draw a bar/histogram

c = categorical({'VLD','ML','DCML'}); % set the x-axis category label
num = [100 800 900]; % set each value
b=bar(c,num,'FaceColor','flat','EdgeColor','none'); % draw histogram/bar chart
b.CData(1,:) = [1 0.37 0.36]; % Set the color of each bar separately
b.CData(2,:) = [0 0.8 0.75];
b.CData(3,:) = [0.45 0.49 0.49];
